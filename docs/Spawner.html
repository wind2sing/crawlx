Spawner

Rules for generating a new Task from url and taskMeta.

x.spawner({
 regex: /forum\/page-\d+/,
 spawn: (url, meta)=>({
   url,
   parse: "title",
   callback(task){
     console.log(task.res.statusCode);
   }
 })
})
